Creamos el script "aumentalista.sh":
#!/bin/bash
inicio=0
fin=19

while [ $inicio -le $fin ]
do
	touch /tmp/varios/coregen-`date +%M:%S:%N`
	inicio=`expr $inicio + 1`
done

rm  -v /tmp/varios/core* 1>> /tmp/listacores
